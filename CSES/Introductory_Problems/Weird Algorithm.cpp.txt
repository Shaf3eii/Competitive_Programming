int lcm(int a, int b)
{
    return (a / gcd(a, b)) * b;
}
 
bool Is_Prime(int p)
{
    if(p < 2)
        return false;
    if(p == 2)
        return true;
    if(p % 2 == 0)
        return false;
 
    int limit = sqrt(p);
    for(int i = 3; i <= limit; i += 2)
    {
        if (p % i == 0)
            return false;
    }
    return true;
}
